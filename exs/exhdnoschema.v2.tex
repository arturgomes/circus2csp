  ResetClock ~\circdef~
  \\\t2 (stime := 0) \circseq Clock\\
\end{circusaction}
\begin{circusaction}
  Clock ~\circdef~
  \\\t2 \circmu~X \circspot 
    ((\circif stime \leq 3 \circthen (tick \then (stime := stime+1) \interleave (getCurrentTime!stime \then \Skip))
        \circelse stime > 2 \circthen ((stime := 0) \circseq \Skip) 
        \circfi) \circseq X)
\end{circusaction}
