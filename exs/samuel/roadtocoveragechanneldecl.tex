 \circchannel acdecl, bcdecl : \nat; ccdecl, dcdecl : \nat \\
\end{circus}

 \circprocess ProcChannelDecl \circdef \circbegin \\
 \circspot acdecl.0 \then bcdecl.0 
      \then \Skip \\
 \circend
\end{circus}
